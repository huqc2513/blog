# webpack基础配置

# 输出多个js

使用glob库进行多文件读取,使用场景，将某个目录下的所有js进行一一单独打包压缩

  let multiEntry = {}
  glob.sync(path.resolve(__dirname, './src/*.js')).forEach(entry => {
    let base = path.basename(entry)
    let ext = path.extname(entry)
    let key = base.replace(ext, '')
    multiEntry[key] = entry
  })

  let entryKeys = Object.keys(multiEntry)
  entryKeys.map((key) => {
      webpackconfig.entry[key] = path.resolve(__dirname, `./src/${key}.js`)
  })

# css转化

针对css或者scss文件解析以及autoprefixer,如果需要将css打包抽离，则用到MiniCssExtractPlugin插件,如果要将css打包到js中，则选用style-loader

      {
        test: /\.(sa|sc|c)ss$/,
        exclude: [
          path.resolve(__dirname, "node_modules")
        ],
        use: [
          {
            loader: MiniCssExtractPlugin.loader,
            options: {
              hmr: process.env.NODE_ENV === 'development',
              reloadAll: true,
            }
          },
          'css-loader',
          {
            loader: "postcss-loader",
            options: {
              plugins: [require('autoprefixer')({
                    browsers: ['last 10 Chrome versions', 'last 5 Firefox versions', 'Safari >= 6', 'ie> 8'] 
                })]
            }
          }
          'sass-loader'
        ]
      }

    //css抽离插件
    plugins: [
        new MiniCssExtractPlugin({
          filename: "css/index.css",
          chunkFilename: "[name].css"
        })

# 图片打包

url-loader使用limit选项将图片转为base64格式减少请求

      {
        test: /\.(png|jpg|jpeg|gif)$/,
        loader: 'url-loader',
        options: {
          outputPath: 'images',
          limit: 1024,  //小于该值的图片会使用base64编码
          name: '[name].[hash:8].[ext]'  //打包后的图片名称 [ext]指图片格式
        }
      }
